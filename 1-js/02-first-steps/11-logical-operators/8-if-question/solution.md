الإجابة: سيتم تنفيذ الأول والثالث

التفاصيل:

```js run
// ينفذ
// نتيجة -1 || 0 = -1 هي نتيجة صادقة
if (-1 || 0) alert("first");

// لا ينفذ
// -1 && 0 = 0, نتيجة كاذبة
if (-1 && 0) alert("second");

// ينفذ
// العامل && له أولوية أعلى من ||
// لذلك يتم تنفيذ -1 && 1 أولا وينتج التالي:
// null || -1 && 1  ->  null || 1  ->  1
if (null || (-1 && 1)) alert("third");
```
